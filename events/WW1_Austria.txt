country_event = {
	id = ww1_austria.5
	title = ww1_austria.5.t
	desc = ww1_austria.5.d
	picture = GFX_report_event_ger_civil_war
	fire_only_once = yes
	trigger = {
		tag = AUS
		has_government = neutrality
		is_subject = no
		date > 1933.5.1
		ITA = {
			has_government = fascism
			is_subject = no
		}
		NOT = {
			has_war_with = ITA
		}
		has_country_leader = {
			character = AUS_engelbert_dollfuss
		}
		has_country_leader_ideology = oligarchism
	}
	mean_time_to_happen = {
		days = 20
	}
	option = {
		name = ww1_austria.5.a
		remove_ideas = nationalism
		set_politics = {
			ruling_party = fascism
			elections_allowed = no
		}
		add_popularity = {
			ideology = fascism
			popularity = 0.15
		}
		set_party_name = {
			ideology = fascism
			long_name = AUS_vaterland_party
			name = AUS_vaterland_party
		}
		AUS_engelbert_dollfuss = {
			promote_character = fascism_ideology
			add_field_marshal_role = {
				traits = {
					media_personality
					politically_connected
					guerilla_fighter
					trait_mountaineer
				}
				skill = 2
				attack_skill = 2
				defense_skill = 1
				planning_skill = 2
				logistics_skill = 2
			}
			set_portraits = {
				civilian = {
					large="gfx/leaders/AUS/AUS_Engelbert_Dollfuss_general.png"
				}
			}
		}
		hidden_effect = {
			promote_character = AUS_wilhelm_miklas
		}
	}
}
