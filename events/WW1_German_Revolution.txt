				}

				target = BAY

				limit = {

					state = 188
				}

					limit = {
						owns_state = 11

country_event = {
	id = freikorps.10
	title = freikorps.10.t
	desc = freikorps.10.d
	picture = GFX_report_event_german_reichstag_gathering
	fire_only_once = yes
	trigger = {
		has_global_flag = german_reunification_wars
		OR = {
			tag = GER
			tag = HAN
			tag = BAY
		}
		has_war = no
		is_subject = no
		is_faction_leader = yes
		all_country = {
			OR = {
				tag = GER
				tag = HAN
				tag = BAY
				tag = SHL
				tag = WUR
				tag = SAX
			}
			OR = {
				tag = PREV
				is_subject_of = PREV
				is_in_faction_with = PREV
				exists = no
			}
		}
	}
	mean_time_to_happen = { days = 0 }
	immediate = {
		hidden_effect = { news_event = freikorps.11 }
	}
	option = {
		name = freikorps.10.a
		set_capital = { state = 64 }
		add_political_power = 200
		every_other_country = {
			limit = {
				OR = {
					tag = GER
					tag = HAN
					tag = BAY
					tag = SHL
					tag = WUR
					tag = SAX
				}
				OR = {
					is_subject_of = PREV
					is_in_faction_with = PREV
				}
			}
			PREV = {
				annex_country = {
					target = PREV

				}

			}
		}
	}
}
