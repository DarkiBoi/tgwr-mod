	id = ww1_siam.5
	title = ww1_siam.5.t
	picture = GFX_report_event_generic_military_parade
	fire_only_once = yes
	trigger = {
		tag = SIA
		has_global_flag = interbellum
		date > 1932.6.1
		is_subject = no
		has_war = no
		has_government = neutrality
	}
	mean_time_to_happen = {
		days = 24
	}
	option = {
		name = ww1_siam.5.a
		set_popularities = {
			neutrality = 40
			democratic = 30
			fascism = 30
		}
		set_politics = {
			ruling_party = fascism
			elections_allowed = no
		}
		set_party_name = {
			ideology = fascism
			long_name = SIA_democratic_party
			name = SIA_democratic_party
		}
		SIA_plaek_phibunsongkhram = {
			add_country_leader_role = {
				promote_leader = yes
				country_leader = {
					ideology = fascism_ideology
				}
			}
			add_field_marshal_role = {
				traits = { politically_connected }
				skill = 3
				attack_skill = 3
				defense_skill = 3
				planning_skill = 2
				logistics_skill = 2
			}
		}
		promote_character = SIA_ananda_mahidol
	}
	option = {
		name = ww1_siam.5.b
		start_civil_war = {
			ruling_party = neutrality
			ideology = fascism
			size = 0.7
			keep_political_party_members = yes

				ideology = fascism

				name = SIA_democratic_party
			}
		}

				set_nationality = THA
			}

				SIA_plaek_phibunsongkhram = {
					add_country_leader_role = {
						promote_leader = yes
						country_leader = {
							ideology = fascism_ideology
						}
					}
					add_field_marshal_role = {
						traits = { politically_connected }
						skill = 3
						attack_skill = 3
						defense_skill = 3
						planning_skill = 2
						logistics_skill = 2
					}
				}
			}
		}
		ai_chance = {
			factor = 0
		}
	}
}
